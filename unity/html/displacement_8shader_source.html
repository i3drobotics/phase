10Shader "I3DR/displacement" {
11 Properties {
12 }
13 SubShader {
14 Tags {
15 "RenderType"="Opaque"
16 }
17 LOD 100
18
19 Pass {
20 CGPROGRAM
21 #pragma vertex vert
22 #pragma fragment frag
23 // We have to target at least 4.5 for StructuredBuffers.
24 #pragma target 4.5
25
26 #include "UnityCG.cginc"
27
28 struct quad_vert
29 {
30 float4 position;
31 float2 corner_coordinates;
32 uint particle_id;
33 };
34
35 struct appdata
36 {
37 uint id : SV_VertexID;
38 };
39
40 struct v2f
41 {
42 float4 vertex : SV_POSITION;
43 half4 color : COLOR0;
44 };
45
46 uniform float particle_size;
47 uniform float near_plane;
48 uniform float far_plane;
49 uniform float2 image_size;
50
51 #ifdef SHADER_API_D3D11
52 uniform StructuredBuffer<quad_vert> particle_buffer;
53 uniform StructuredBuffer<uint> color_buffer;
54 uniform StructuredBuffer<float> depth_buffer;
55 #endif
56
57 // Unpack a 32-bit int into colour components.
58 fixed4 uint_to_fixed4(uint a)
59 {
60 fixed4 result;
61 //result.r = GammaToLinearSpace(((a >> 0) & 0xFF) / 255.0);
62 //result.g = GammaToLinearSpace(((a >> 8) & 0xFF) / 255.0);
63 //result.b = GammaToLinearSpace(((a >> 16) & 0xFF) / 255.0);
64 result.r = ((a >> 0) & 0xFF) / 255.0;
65 result.g = ((a >> 8) & 0xFF) / 255.0;
66 result.b = ((a >> 16) & 0xFF) / 255.0;
67 result.a = ((a >> 24) & 0xFF) / 255.0;
68 return result;
69 }
70
71 // Calculate the index into an array given its x and y coordinates.
72 uint IndexToID(uint2 index)
73 {
74 return index.y * image_size.x + index.x;
75 }
76
77 // Calculate the x and y coordinates given an array index.
78 uint2 IDToIndex(uint id)
79 {
80 return uint2(id % image_size.x, id / image_size.x);
81 }
82
83 v2f vert(const appdata v)
84 {
85 v2f o;
86
87 quad_vert qv = particle_buffer[v.id];
88
89 const float4 position = qv.position;
90
91 // Get the distance of the particle from the render camera to calculate the screen-space size in pixels.
92 const float depth = UnityWorldToViewPos(position).z;
93 const float2 size = float2(depth / _ScreenParams.x, depth / _ScreenParams.y) * particle_size;
94
95 // Project the particle to screen-space, expanding out the corners of each particle.
96 o.vertex = UnityWorldToClipPos(position) + float4(qv.corner_coordinates * size, 0.0, 0.0);
97
98 o.color = uint_to_fixed4(color_buffer[qv.particle_id]);
99
100 // Near/far clipping.
101 if (depth_buffer[qv.particle_id] > far_plane || depth_buffer[qv.particle_id] < near_plane)
102 {
103 // We use an alpha of 0 to indicate the particle mustn't be drawn.
104 o.color.a = 0.0;
105 }
106
107 return o;
108 }
109
110 fixed4 frag(v2f i) : SV_Target
111 {
112 // Discard this fragment if it's not to be drawn. This isn't super-efficient and we should eventually
113 // contrive a way to cull it in the compute shader, way before it gets here.
114 if (i.color.a == 0.0)
115 {
116 discard;
117 }
118
119 return i.color;
120 }
121 ENDCG
122 }
123 }
