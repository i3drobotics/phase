271 private void UpdateCameraParameters()
272 {
273 float tz = _ZFar - _ZNear;
274 float ty = Mathf.Tan(_verticalFOV * Mathf.Deg2Rad / 2.0f) * _ZFar * 2.0f;
275 float tx = ty * _aspectRatio;
276
277 // The axis-aligned bounding box actually fits the frustum's oriented bounding box, so it tends to over
278 // estimate the size of the box. But as this is just used by Unity for culling the whole cloud, it's not
279 // a big issue.
280 Vector3 centre = transform.TransformPoint(0.0f, 0.0f, (_ZFar - _ZNear) / 2.0f + _ZNear);
281 Vector3[] corners = {
282 transform.TransformVector(tx, ty, tz),
283 transform.TransformVector(tx, ty, -tz),
284 transform.TransformVector(tx, -ty, tz),
285 transform.TransformVector(tx, -ty, -tz),
286 transform.TransformVector(-tx, ty, tz),
287 transform.TransformVector(-tx, ty, -tz),
288 transform.TransformVector(-tx, -ty, tz),
289 transform.TransformVector(-tx, -ty, -tz)
290 };
291
292 Vector3 extents = new Vector3();
293
294 for (int i = 0; i < 8; ++i)
295 {
296 extents.x = Mathf.Max(extents.x, corners[i].x);
297 extents.y = Mathf.Max(extents.y, corners[i].y);
298 extents.z = Mathf.Max(extents.z, corners[i].z);
299 }
300
301 _boundingBox = new Bounds(centre, extents);
302
303 Matrix4x4 proj = Matrix4x4.Perspective(_verticalFOV, _aspectRatio, _ZNear, _ZFar);
304 // Used to make the projection matrix work regardless of rendering API.
305 _cameraTransform = GL.GetGPUProjectionMatrix(proj, false);
306 }

316 _computeShader.Dispatch(_displacementKernel, (_imageWidth + 7) / 8, (_imageHeight + 7) / 8, 1);
317 }
