11 Shader "I3DR/displacement" {
12  Properties {
13  }
14  SubShader {
15  Tags {
16  "RenderType"="Opaque"
17  }
18  LOD 100
19 
20  Pass {
21  CGPROGRAM
22  #pragma vertex vert
23  #pragma fragment frag
24  // We have to target at least 4.5 for StructuredBuffers.
25  #pragma target 4.5
26 
27  #include "UnityCG.cginc"
28 
29  struct quad_vert
30  {
31  float4 position;
32  float2 corner_coordinates;
33  uint particle_id;
34  };
35 
36  struct appdata
37  {
38  uint id : SV_VertexID;
39  };
40 
41  struct v2f
42  {
43  float4 vertex : SV_POSITION;
44  half4 color : COLOR0;
45  };
46 
47  uniform float particle_size;
48  uniform float near_plane;
49  uniform float far_plane;
50  uniform float2 image_size;
51 
52  #ifdef SHADER_API_D3D11
53  uniform StructuredBuffer<quad_vert> particle_buffer;
54  uniform StructuredBuffer<uint> color_buffer;
55  uniform StructuredBuffer<float> depth_buffer;
56  #endif
57 
58  // Unpack a 32-bit int into colour components.
59  fixed4 uint_to_fixed4(uint a)
60  {
61  fixed4 result;
62  //result.r = GammaToLinearSpace(((a >> 0) & 0xFF) / 255.0);
63  //result.g = GammaToLinearSpace(((a >> 8) & 0xFF) / 255.0);
64  //result.b = GammaToLinearSpace(((a >> 16) & 0xFF) / 255.0);
65  result.r = ((a >> 0) & 0xFF) / 255.0;
66  result.g = ((a >> 8) & 0xFF) / 255.0;
67  result.b = ((a >> 16) & 0xFF) / 255.0;
68  result.a = ((a >> 24) & 0xFF) / 255.0;
69  return result;
70  }
71 
72  // Calculate the index into an array given its x and y coordinates.
73  uint IndexToID(uint2 index)
74  {
75  return index.y * image_size.x + index.x;
76  }
77 
78  // Calculate the x and y coordinates given an array index.
79  uint2 IDToIndex(uint id)
80  {
81  return uint2(id % image_size.x, id / image_size.x);
82  }
83 
84  v2f vert(const appdata v)
85  {
86  v2f o;
87 
88  quad_vert qv = particle_buffer[v.id];
89 
90  const float4 position = qv.position;
91 
92  // Get the distance of the particle from the render camera to calculate the screen-space size in pixels.
93  const float depth = UnityWorldToViewPos(position).z;
94  const float2 size = float2(depth / _ScreenParams.x, depth / _ScreenParams.y) * particle_size;
95 
96  // Project the particle to screen-space, expanding out the corners of each particle.
97  o.vertex = UnityWorldToClipPos(position) + float4(qv.corner_coordinates * size, 0.0, 0.0);
98 
99  o.color = uint_to_fixed4(color_buffer[qv.particle_id]);
100 
101  // Near/far clipping.
102  if (depth_buffer[qv.particle_id] > far_plane || depth_buffer[qv.particle_id] < near_plane)
103  {
104  // We use an alpha of 0 to indicate the particle mustn't be drawn.
105  o.color.a = 0.0;
106  }
107 
108  return o;
109  }
110 
111  fixed4 frag(v2f i) : SV_Target
112  {
113  // Discard this fragment if it's not to be drawn. This isn't super-efficient and we should eventually
114  // contrive a way to cull it in the compute shader, way before it gets here.
115  if (i.color.a == 0.0)
116  {
117  discard;
118  }
119 
120  return i.color;
121  }
122  ENDCG
123  }
124  }
125 }
