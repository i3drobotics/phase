11 #pragma kernel GenerateParticles
12 
13 struct quad_vert
14 {
15  float4 position;
16  float2 corner_coordinates;
17  uint particle_id;
18 };
19 
20 uniform StructuredBuffer<float> depth_buffer;
21 uniform RWStructuredBuffer<quad_vert> particle_buffer;
22 uniform float4x4 camera_p;
23 uniform float4x4 camera_p_inv;
24 uniform float4x4 camera_v_inv;
25 uniform float2 image_size;
26 uniform int flip_x;
27 uniform int flip_y;
28 uniform float scale;
29 
30 [numthreads(8, 8, 1)]
31 void GenerateParticles(uint3 id : SV_DispatchThreadID)
32 {
33  // If we're working on an image who's dimensions are not multiples of the thread count (8,8 currently), skip it.
34  if (id.x >= (uint)image_size.x || id.y >= (uint)image_size.y)
35  {
36  return;
37  }
38 
39  // Convert the depth value from local space to clip space, to put the final coordinate in clip space.
40  const uint buf_idx = id.y * image_size.x + id.x;
41  const float z = depth_buffer[buf_idx];
42  const float x = id.x / image_size.x * 2.0 - 1.0;
43  const float y = id.y / image_size.y * 2.0 - 1.0;
44  float4 clip_pos = mul(camera_p, float4(0.0, 0.0, z, 1.0));
45  clip_pos.x = x * -flip_x;
46  clip_pos.y = y * flip_y;
47  clip_pos.wz /= clip_pos.w;
48 
49  // Unproject the point from clip space back to local space, apply scaling, then reproject into world space.
50  const float4 homogenous_local_pos = mul(camera_p_inv, clip_pos);
51  const float4 local_pos = (homogenous_local_pos / homogenous_local_pos.w) * float4(scale.xxx, 1.0);
52  const float4 world_pos = mul(camera_v_inv, local_pos);
53 
54  // Build each vertex of the quad. We don't know where each corner will be yet, so set them all to the same world
55  // position, and mark which corner is which for later scaling in the render shader.
56  quad_vert v1;
57  v1.position = world_pos;
58  v1.corner_coordinates = float2(1.0, 1.0);
59  v1.particle_id = buf_idx;
60  particle_buffer[buf_idx * 4 + 0] = v1;
61 
62  quad_vert v2;
63  v2.position = world_pos;
64  v2.corner_coordinates = float2(-1.0, 1.0);
65  v2.particle_id = buf_idx;
66  particle_buffer[buf_idx * 4 + 1] = v2;
67 
68  quad_vert v3;
69  v3.position = world_pos;
70  v3.corner_coordinates = float2(-1.0, -1.0);
71  v3.particle_id = buf_idx;
72  particle_buffer[buf_idx * 4 + 2] = v3;
73 
74  quad_vert v4;
75  v4.position = world_pos;
76  v4.corner_coordinates = float2(1.0, -1.0);
77  v4.particle_id = buf_idx;
78  particle_buffer[buf_idx * 4 + 3] = v4;
79 }
