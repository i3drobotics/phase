160  void processReadResult(StereoVisionReadResult readResult)
161  {
162  if (readResult.valid)
163  {
164  Debug.Log("New frame");
165  int imageWidth = sv.getWidth();
166  int imageHeight = sv.getHeight();
167  if (LeftImageTexture == null || LeftImageTexture.width != imageWidth || LeftImageTexture.height != imageHeight)
168  {
169  LeftImageTexture = new Texture2D(imageWidth, imageHeight, TextureFormat.RGBA32, false);
170  LeftImageMaterial.mainTexture = LeftImageTexture;
171  }
172  if (RightImageTexture == null || RightImageTexture.width != imageWidth || RightImageTexture.height != imageHeight)
173  {
174  RightImageTexture = new Texture2D(imageWidth, imageHeight, TextureFormat.RGBA32, false);
175  RightImageMaterial.mainTexture = RightImageTexture;
176  }
177  if (DepthImageTexture == null || DepthImageTexture.width != imageWidth || DepthImageTexture.height != imageHeight)
178  {
179  DepthImageTexture = new Texture2D(imageWidth, imageHeight, TextureFormat.RGBA32, false);
180  DepthImageMaterial.mainTexture = DepthImageTexture;
181  }
182 
183  // get z only from depth
184  //StereoSupport.depth2depthz(readResult.depth, imageWidth, imageHeight, depth_z);
185  depthRenderer.UpdateBuffers(readResult.left_image, readResult.depth);
186 
187  LeftImageTexture.LoadRawTextureData(readResult.left_image);
188  LeftImageTexture.Apply(false);
189 
190  RightImageTexture.LoadRawTextureData(readResult.right_image);
191  RightImageTexture.Apply(false);
192 
193  StereoSupport.normaliseDisparity(readResult.disparity, imageWidth, imageHeight, disp_image);
194  // convert BGRA colormap to RGBA
195  // StereoSupport.bgra2rgba(disp_image, imageWidth, imageHeight, disp_image);
196  DepthImageTexture.LoadRawTextureData(disp_image);
197  DepthImageTexture.Apply(false);
