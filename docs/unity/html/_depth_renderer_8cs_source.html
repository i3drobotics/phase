272  private void UpdateCameraParameters()
273  {
274  float tz = _ZFar - _ZNear;
275  float ty = Mathf.Tan(_verticalFOV * Mathf.Deg2Rad / 2.0f) * _ZFar * 2.0f;
276  float tx = ty * _aspectRatio;
277 
278  // The axis-aligned bounding box actually fits the frustum's oriented bounding box, so it tends to over
279  // estimate the size of the box. But as this is just used by Unity for culling the whole cloud, it's not
280  // a big issue.
281  Vector3 centre = transform.TransformPoint(0.0f, 0.0f, (_ZFar - _ZNear) / 2.0f + _ZNear);
282  Vector3[] corners = {
283  transform.TransformVector(tx, ty, tz),
284  transform.TransformVector(tx, ty, -tz),
285  transform.TransformVector(tx, -ty, tz),
286  transform.TransformVector(tx, -ty, -tz),
287  transform.TransformVector(-tx, ty, tz),
288  transform.TransformVector(-tx, ty, -tz),
289  transform.TransformVector(-tx, -ty, tz),
290  transform.TransformVector(-tx, -ty, -tz)
291  };
292 
293  Vector3 extents = new Vector3();
294 
295  for (int i = 0; i < 8; ++i)
296  {
297  extents.x = Mathf.Max(extents.x, corners[i].x);
298  extents.y = Mathf.Max(extents.y, corners[i].y);
299  extents.z = Mathf.Max(extents.z, corners[i].z);
300  }
301 
302  _boundingBox = new Bounds(centre, extents);
303 
304  Matrix4x4 proj = Matrix4x4.Perspective(_verticalFOV, _aspectRatio, _ZNear, _ZFar);
305  // Used to make the projection matrix work regardless of rendering API.
306  _cameraTransform = GL.GetGPUProjectionMatrix(proj, false);
307  }
