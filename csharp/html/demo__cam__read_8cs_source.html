58 class DemoCameraRead
59 {
60 static int Main(string[] args)
61 {
62 ImageViewer viewer_disparity = new ImageViewer();
63 ImageViewer viewer_left = new ImageViewer();
64 ImageViewer viewer_right = new ImageViewer();
65
66 bool license_valid = StereoI3DRSGM.isLicenseValid();
67 if (license_valid){
68 Console.WriteLine("I3DRSGM license accepted");
69 } else {
70 Console.WriteLine("Missing or invalid I3DRSGM license");
71 }
72 string resource_folder = "resources";
73 string camera_name = "stereotheatresim";
74 string left_serial = "0815-0000";
75 string right_serial = "0815-0001";
76 string cal_type = "ros"; // ros / cv
77 CameraDeviceType dev_type = CameraDeviceType.DEVICE_TYPE_GENERIC_PYLON; // DEVICE_TYPE_GENERIC_PYLON / DEVICE_TYPE_TITANIA
78 bool use_test_images = true;
79 int repeat_capture = 1;
80 bool show_display = true;
81 int disp_time = 1;
82
83 CameraInterfaceType interface_type;
84 if (use_test_images){
85 interface_type = CameraInterfaceType.INTERFACE_TYPE_VIRTUAL;
86 } else {
87 interface_type = CameraInterfaceType.INTERFACE_TYPE_USB;
88 }
89
90 string left_yaml = resource_folder + "/test/" + camera_name + "/" + cal_type + "/left.yaml";
91 string right_yaml = resource_folder + "/test/" + camera_name + "/" + cal_type + "/right.yaml";
92
93 CameraDeviceInfo device_info = new CameraDeviceInfo(
94 left_serial, right_serial, camera_name,
95 dev_type, interface_type
96 );
97
98 StereoMatcherType matcher_type;
99 if (license_valid){
100 matcher_type = StereoMatcherType.STEREO_MATCHER_I3DRSGM;
101 } else {
102 matcher_type = StereoMatcherType.STEREO_MATCHER_BM;
103 }
104
105 StereoVision sv = new StereoVision(device_info, matcher_type, left_yaml, right_yaml);
106 if (!sv.isValidCalibration()) {
107 Console.WriteLine("Failed to load calibration");
108 sv.dispose();
109 return 1;
110 }
111
112 Console.WriteLine("StereoVision instance created");
113 if (use_test_images)
114 {
115 sv.setTestImagePaths(
116 resource_folder + "/test/" + camera_name + "/left.png",
117 resource_folder + "/test/" + camera_name + "/right.png"
118 );
119 }
120
121 AbstractStereoMatcher matcher;
122 sv.getMatcher(out matcher);
123 if (matcher is StereoI3DRSGM){
124 Console.WriteLine("Is I3DRSGM");
125 ((StereoI3DRSGM)matcher).setWindowSize(3);
126 ((StereoI3DRSGM)matcher).setNumDisparities(49);
127 ((StereoI3DRSGM)matcher).setSpeckleMaxSize(1000);
128 ((StereoI3DRSGM)matcher).setSpeckleMaxDiff(0.5f);
129 }
130
131 Console.WriteLine("Connecting to camera...");
132 bool ret = sv.connect();
133
134 if (ret){
135 Console.WriteLine("Camera connected");
136 int disp_width = (int)((float)sv.getWidth() * 0.25f);
137 int disp_height = (int)((float)sv.getHeight() * 0.25f);
138 int disp_stride = disp_width * 4;
139 Console.WriteLine("Running non-threaded camera capture...");
140 for (int i = 0; i < repeat_capture; i++)
141 {
142 Console.WriteLine("Waiting for result...");
143 StereoVisionReadResult read_result = sv.read();
144 if (read_result.valid)
145 {
146 Console.WriteLine("Stereo result received");
147 //float val = read_result.disparity[sv.getWidth()*r+nchannels*c+ch]; //useful to remember how to access array elements
148 byte[] disp_left = Utils.scaleImage(read_result.left_image, sv.getWidth(), sv.getHeight(), 0.25f);
149 byte[] disp_right = Utils.scaleImage(read_result.right_image, sv.getWidth(), sv.getHeight(), 0.25f);
150 byte[] norm_disparity = Utils.normaliseDisparity(read_result.disparity, sv.getWidth(), sv.getHeight());
151 byte[] disp_disparity = Utils.scaleImage(norm_disparity, sv.getWidth(), sv.getHeight(), 0.25f);
152 if (show_display) {
153 byte[] disp_left_rgba = Utils.bgr2bgra(disp_left, disp_width, disp_height);
154 byte[] disp_right_rgba = Utils.bgr2bgra(disp_right, disp_width, disp_height);
155 byte[] disp_disparity_rgba = Utils.bgr2bgra(disp_disparity, disp_width, disp_height);
156 viewer_left.setImage(disp_left_rgba, disp_width, disp_height, disp_stride);
157 viewer_right.setImage(disp_right_rgba, disp_width, disp_height, disp_stride);
158 viewer_disparity.setImage(disp_disparity_rgba, disp_width, disp_height, disp_stride);
159 viewer_left.show(disp_time);
160 viewer_right.show(disp_time);
161 viewer_disparity.show(disp_time);
162 }
163 }
164 else
165 {
166 Console.WriteLine("Failed to read stereo result");
167 }
168 }
169
170 sv.disconnect();
171
172 Console.WriteLine("Camera disconnected");
173 ret = sv.connect();
174 if (ret)
175 {
176 AbstractStereoCamera cam;
177 StereoCameraCalibration cal;
178 sv.getCamera(out cam);
179 sv.getCalibration(out cal);
180 int width = sv.getWidth();
181 int height = sv.getHeight();
182 Console.WriteLine("Running split threaded camera capture...");
183 for (int i = 0; i < repeat_capture; i++)
184 {
185 cam.startReadThread();
186 Console.WriteLine("Waiting for result...");
187 while (cam.isReadThreadRunning()) { }
188 CameraReadResult cam_result = cam.getReadThreadResult();
189 if (cam_result.valid)
190 {
191 StereoImagePair rect_image_pair = cal.rectify(
192 cam_result.left_image, cam_result.right_image,
193 width, height
194 );
195 Console.WriteLine("Starting compute...");
196 matcher.startComputeThread(rect_image_pair.left, rect_image_pair.right, width, height);
197 Console.WriteLine("Stereo result received");
198 Console.WriteLine("Framerate: {0}", cam.getFrameRate());
199 byte[] disp_left = Utils.scaleImage(rect_image_pair.left, width, height, 0.25f);
200 byte[] disp_right = Utils.scaleImage(rect_image_pair.right, width, height, 0.25f);
201 if (show_display)
202 {
203 byte[] disp_left_rgba = Utils.bgr2bgra(disp_left, disp_width, disp_height);
204 byte[] disp_right_rgba = Utils.bgr2bgra(disp_right, disp_width, disp_height);
205 viewer_left.setImage(disp_left_rgba, disp_width, disp_height, disp_stride);
206 viewer_right.setImage(disp_right_rgba, disp_width, disp_height, disp_stride);
207 viewer_left.show(disp_time);
208 viewer_right.show(disp_time);
209 }
210 Console.WriteLine("Waiting for result...");
211 while (matcher.isComputeThreadRunning()) { }
212 StereoMatcherComputeResult match_result = matcher.getComputeThreadResult(width, height);
213 if (match_result.valid)
214 {
215 Console.WriteLine("Match result received");
216 byte[] norm_disparity = Utils.normaliseDisparity(match_result.disparity, width, height);
217 byte[] disp_disparity = Utils.scaleImage(norm_disparity, width, height, 0.25f);
218 if (show_display)
219 {
220 byte[] disp_disparity_rgba = Utils.bgr2bgra(disp_disparity, disp_width, disp_height);
221 viewer_disparity.setImage(disp_disparity_rgba, disp_width, disp_height, disp_stride);
222 viewer_disparity.show(disp_time);
223 }
224 }
225 else
226 {
227 Console.WriteLine("Failed to compute stereo match");
228 }
229 }
230 else
231 {
232 Console.WriteLine("Failed to read stereo result");
233 }
234 }
235
236 Console.WriteLine("Running threaded camera capture...");
237 for (int i = 0; i < repeat_capture; i++)
238 {
239 sv.startReadThread();
240 Console.WriteLine("Waiting for result...");
241 while (sv.isReadThreadRunning()) { }
242 StereoVisionReadResult result = sv.getReadThreadResult();
243 if (result.valid)
244 {
245 Console.WriteLine("Stereo result received");
246 Console.WriteLine("Framerate: {0}", cam.getFrameRate());
247 byte[] disp_left = Utils.scaleImage(result.left_image, width, height, 0.25f);
248 byte[] disp_right = Utils.scaleImage(result.right_image, width, height, 0.25f);
249 byte[] norm_disparity = Utils.normaliseDisparity(result.disparity, width, height);
250 byte[] disp_disparity = Utils.scaleImage(norm_disparity, width, height, 0.25f);
251 if (show_display)
252 {
253 byte[] disp_left_rgba = Utils.bgr2bgra(disp_left, disp_width, disp_height);
254 byte[] disp_right_rgba = Utils.bgr2bgra(disp_right, disp_width, disp_height);
255 byte[] disp_disparity_rgba = Utils.bgr2bgra(disp_disparity, disp_width, disp_height);
256 viewer_left.setImage(disp_left_rgba, disp_width, disp_height, disp_stride);
257 viewer_right.setImage(disp_right_rgba, disp_width, disp_height, disp_stride);
258 viewer_disparity.setImage(disp_disparity_rgba, disp_width, disp_height, disp_stride);
259 viewer_left.show(disp_time);
260 viewer_right.show(disp_time);
261 viewer_disparity.show(disp_time);
262 }
263 }
264 else
265 {
266 Console.WriteLine("Failed to read stereo result");
267 }
268 }
269 }
270 sv.disconnect();
271 Console.WriteLine("Camera disconnected");
272 }
273 sv.dispose();
274 return 0;
275 }
276 }
