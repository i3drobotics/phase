58 class DemoRGBD
59 {
60 static int Main(string[] args)
61 {
62 string camera_name = "stereotheatresim";
63 string cal_type = "ros";
64#if PHASE_DEBUG
65 string out_folder = "../../out/csharp";
66 string resource_folder = "../../resources";
67#else
68 string out_folder = "out/csharp";
69 string resource_folder = "resources";
70#endif
71 string left_yaml = resource_folder + "/test/" + camera_name + "/" + cal_type + "/left.yaml";
72 string right_yaml = resource_folder + "/test/" + camera_name + "/" + cal_type + "/right.yaml";
73 string left_image_file = resource_folder + "/test/" + camera_name + "/left.png";
74 string right_image_file = resource_folder + "/test/" + camera_name + "/right.png";
75 string out_rgb_video = out_folder + "/rgb.mp4";
76 string out_depth_video = out_folder + "/depth.avi";
77
78 int num_of_frames = 1;
79
80 Console.WriteLine("Loading test images...");
81 //TODO get image size from file
82 int image_width = 2448;
83 int image_height = 2048;
84 byte[] left_image_cv = Utils.readImage(left_image_file, image_width, image_height);
85 byte[] right_image_cv = Utils.readImage(right_image_file, image_width, image_height);
86
87 if (left_image_cv.Length == 0) { return 1; }
88 if (right_image_cv.Length == 0) { return 1; }
89
90 StereoCameraCalibration calibration = StereoCameraCalibration.calibrationFromYAML(left_yaml, right_yaml);
91
92 if (!calibration.isValid()) { return 1; }
93
94 StereoImagePair rect_image_pair = calibration.rectify(left_image_cv, right_image_cv, image_width, image_height);
95
96 int image_rows = 2048;
97 int image_cols = 2448;
98 int image_channels = 3;
99
100 MatrixUInt8 left_image = new MatrixUInt8(
101 image_rows, image_cols, image_channels,
102 rect_image_pair.left, true);
103 MatrixUInt8 right_image = new MatrixUInt8(
104 image_rows, image_cols, image_channels,
105 rect_image_pair.right, true);
106
107 Console.WriteLine("Processing stereo...");
108 StereoParams stereo_params = new StereoParams(
109 StereoMatcherType.STEREO_MATCHER_BM,
110 11, 0, 25, false
111 );
112 MatrixFloat disparity = StereoProcess.processStereo(
113 stereo_params, left_image, right_image, calibration, false
114 );
115
116 if (disparity.isEmpty()) { return 1; }
117
118 float[] disparity_cv = new float[disparity.getLength()];
119 disparity_cv = disparity.getData();
120
121 float[] depth = Utils.disparity2Depth(disparity_cv, image_width, image_height, calibration.getQ());
122
123 if (depth.Length == 0) { return 1; }
124
125 Console.WriteLine("Setting up video writing...");
126 RGBDVideoWriter rgbdVideoWriter = new RGBDVideoWriter(
127 out_rgb_video, out_depth_video,
128 left_image.getColumns(), left_image.getRows()
129 );
130 if (!rgbdVideoWriter.isOpened()) { return 1; }
131 Console.WriteLine("Writing video...");
132 for (int i = 0; i < num_of_frames; i++)
133 {
134 rgbdVideoWriter.add(rect_image_pair.left, depth);
135 }
136
137 Console.WriteLine("Saving video file...");
138 rgbdVideoWriter.saveThreaded();
139 while (rgbdVideoWriter.isSaveThreadRunning()) { }
140 if (!rgbdVideoWriter.getSaveThreadResult()) { return 1; }
141
142 Console.WriteLine("Loading video stream...");
143 RGBDVideoStream rgbdVideoStream = new RGBDVideoStream(
144 out_rgb_video, out_depth_video,
145 left_image.getColumns(), left_image.getRows()
146 );
147 if (!rgbdVideoStream.isOpened()) { return 1; }
148 rgbdVideoStream.loadThreaded();
149 while (rgbdVideoStream.isLoadThreadRunning()) { };
150 if (!rgbdVideoStream.getLoadThreadResult()) { return 1; }
151
152 float[] depth_sim_input = new float[disparity.getLength()];
153 depth.CopyTo(depth_sim_input, 0);
154
155 for (int i = 0; i < disparity.getRows(); i++)
156 {
157 for (int j = 0; j < disparity.getColumns(); j++)
158 {
159 float val = depth_sim_input[i * disparity.getColumns() + j];
160 val *= 6553.5f;
161 // TODO replace Math.Round here as it is very slow
162 ushort sval = (ushort)(Math.Round(val));
163 float fval = (float)sval;
164 fval *= (1.0f / 6553.5f);
165 depth_sim_input[i * disparity.getColumns() + j] = fval;
166 }
167 }
168
169 float min_resolv = 10.0f / 65535.0f;
170
171 Console.WriteLine("Streaming video...");
172 while (!rgbdVideoStream.isFinished())
173 {
174 RGBDVideoFrame frame = rgbdVideoStream.read();
175 for (int i = 0; i < disparity.getRows(); i++)
176 {
177 for (int j = 0; j < disparity.getColumns(); j++)
178 {
179 float in_val = depth_sim_input[i * disparity.getColumns() + j];
180 float out_val = frame.depth[i * disparity.getColumns() + j];
181 float diff = Math.Abs(in_val - out_val);
182 if (diff > min_resolv) {
183 Console.WriteLine(String.Format("RGBD video depth value difference from origional is {0} which is greater miniumum {1}", diff, min_resolv));
184 return 1;
185 }
186 }
187 }
188 if (!Utils.cvMatIsEqual(frame.depth, depth_sim_input, image_width, image_height, 1)){
189 Console.WriteLine(String.Format("Loaded RGBD video depth image differs from origional"));
190 rgbdVideoStream.close();
191 return 1;
192 }
193 }
194 Console.WriteLine("Closing video stream...");
195 rgbdVideoStream.close();
196
197 Console.WriteLine("Done.");
198
199 rgbdVideoWriter.dispose();
200 rgbdVideoStream.dispose();
201
202 return 0;
203 }
204 }
205 }
